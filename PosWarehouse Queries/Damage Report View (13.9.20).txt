DROP VIEW POSWAREHOUSE.VEW_RPT_DAMAGE_DETAILS;


CREATE OR REPLACE FORCE VIEW POSWAREHOUSE.VEW_RPT_DAMAGE_DETAILS
(
   RPT_TITLE,
   DAMAGE_TRANSFER_ITEM_ID,
   DAMAGE_TRANSFER_CHALLAN_NUM,
   CREATED_BY,
   CREATED_DATE,
   APPROVED_BY,
   APPROVED_DATE,
   DAMAGE_TRANSFER_ID,
   ITEM_ID,
   PRODUCT_ID,
   ITEM_NAME,
   BARCODE,
   TRANSFER_QUANTITY,
   SALE_PRICE,
   REMARKS
)
AS
   SELECT '' RPT_TITLE,
          I.DAMAGE_TRANSFER_ITEM_ID,
          M.DAMAGE_TRANSFER_CHALLAN_NUM,
          M.CREATED_BY,
          M.CREATED_DATE,
          M.APPROVED_BY,
          M.APPROVED_DATE,
          I.DAMAGE_TRANSFER_ID,
          I.ITEM_ID,
          I.PRODUCT_ID,
          I.ITEM_NAME,
          I.BARCODE,
          I.TRANSFER_QUANTITY,
          I.SALE_PRICE,
          I.REMARKS
     FROM    DAMAGE_TRANSFER_ITEM I
          LEFT JOIN
             DAMAGE_TRANSFER_MAIN M
          ON M.DAMAGE_TRANSFER_ID = I.DAMAGE_TRANSFER_ID;
