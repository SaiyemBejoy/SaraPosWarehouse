CREATE OR REPLACE FORCE VIEW POSWAREHOUSE.VEW_REQUISITION_MAIN_ITEM
(
   REQUISITION_MAIN_ITEM_ID,
   REQUISITION_ID,
   BARCODE,
   ITEM_NAME,
   PRICE,
   QUANTITY
)
AS
   SELECT REQUISITION_MAIN_ITEM_ID,
          REQUISITION_ID,
          BARCODE,
          ITEM_NAME,
          PRICE,
          QUANTITY
     FROM REQUISITION_MAIN_ITEM;
